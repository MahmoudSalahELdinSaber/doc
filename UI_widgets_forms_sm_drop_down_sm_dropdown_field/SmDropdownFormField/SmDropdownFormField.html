SmDropdownFormField(
    {Key? key,
    InputDecoration? decoration,
    T? initialValue,
    required List<DropdownMenuItem<T>>? items,
    AutovalidateMode autoValidate = AutovalidateMode.disabled,
    FormFieldSetter<T>? onSaved,
    FormFieldValidator<T>? validator,
    required BuildContext context,
    String? hintText,
    TextStyle? hintStyle,
    FocusNode? focusNode,
    this.dropdownFormFieldStyle = const DropdownFormFieldStyle(),
    this.borderColor,
    this.dropDownValue = '',
    this.enabledBorder,
    this.borderRadius = const BorderRadius.all(
      const Radius.circular(50.0),
    ),
    this.focusedBorder,
    this.enable = true,
    this.errorBorder,
    this.focusedErrorBorder,
    this.contentPadding,
    this.onChanged,
    this.onTap,
    this.icon = const Icon(Icons.arrow_drop_down),
    this.disabledColor,
    this.dropDownColor,
    this.iconSize,
    this.textStyle})
    : super(
        key: key,
        onSaved: onSaved,
        validator: validator,
        autovalidateMode: autoValidate,
        initialValue: items != null && items.contains(initialValue)
            ? initialValue
            : (initialValue != null)
                ? initialValue
                : null,
        builder: (FormFieldState<T> field) {
          final InputDecoration effectiveDecoration = (decoration ??
                  const InputDecoration())
              .applyDefaults(Theme.of(field.context).inputDecorationTheme);
          return InputDecorator(
            decoration: effectiveDecoration.copyWith(
                errorText: field.hasError ? field.errorText : null,
                fillColor:
                    enable ? effectiveDecoration.fillColor : disabledColor,
                filled: true,
                counterText: "",
                enabled: enable,
                contentPadding: contentPadding == null
                    ? const EdgeInsets.only(
                        top: 11, bottom: 12, left: 16, right: 16)
                    : contentPadding,
                enabledBorder: enabledBorder,
                focusedBorder: focusedBorder,
                disabledBorder: enabledBorder,
                errorBorder: errorBorder,
                focusedErrorBorder: focusedErrorBorder),
            isEmpty: field.value == '' || field.value == null,
            child: DropdownButtonHideUnderline(
              child: ButtonTheme(
                alignedDropdown: false,
                child: DropdownButton<T>(
                  isExpanded: true,
                  style: textStyle ?? textColor1Text_14.textStyle,
                  icon: icon,
                  iconEnabledColor: dropdownFormFieldStyle.iconColor,
                  iconDisabledColor: dropdownFormFieldStyle.iconColor,
                  iconSize: iconSize ?? 20,
                  dropdownColor: dropDownColor,
                  value: enable
                      ? dropDownValue == ''
                          ? field.value
                          : dropDownValue ?? initialValue
                      : initialValue,
                  isDense: true,
                  hint: Text(
                    hintText ?? '',
                    style: hintStyle ??
                        TextStyle(color: uiTheme.onBackgroundColor),
                  ),
                  onTap: () {
                    if (onTap != null) onTap();
                  },
                  onChanged: enable
                      ? (T? value) {
                          field.didChange(value);
                          onChanged != null ? onChanged(value) : () {};
                        }
                      : null,
                  focusNode: focusNode,
                  items: items?.toList(),
                  menuMaxHeight: SizeConfig(context).screenBodyHeight,
                ),
              ),
            ),
          );
        },
      );
