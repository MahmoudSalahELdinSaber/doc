ChartSeries createCandleStick() {
  return CandleSeries<SmChartDataModel, dynamic>(
      name: chartCollectionModel!.chartSeriesName,
      isVisible: chartCollectionModel!.isVisible,
      dataSource: chartCollectionModel!.dataSource,
      bullColor: this.style!.candleStickBearColor,
      bearColor: this.style!.candleStickBearColor,
      xValueMapper: (SmChartDataModel model, _) => model.xValue,
      lowValueMapper: (SmChartDataModel model, _) => model.lowValue,
      highValueMapper: (SmChartDataModel model, _) => model.highValue,
      openValueMapper: (SmChartDataModel model, _) => model.openValue,
      closeValueMapper: (SmChartDataModel model, _) => model.closeValue);
}
