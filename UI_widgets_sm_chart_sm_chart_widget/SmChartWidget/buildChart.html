Widget buildChart(BuildContext context,
    {bool isLoading = false, bool enableFullScreen = false}) {
  // customChart.reset();
  return Stack(
    children: [
      Positioned(
          child: enableFullScreen
              ? _chartWidgetState.buildChartWidget(false, isLoading,
                  isFullScreenMode: true)
              : _chartWidgetState.expandedWidget(
                  context,
                  isLoading: isLoading,
                )),
      Positioned(
          top: 0,
          left: 0,
          right: 0,
          bottom: 0,
          child: Center(
              child: CircularProgressIndicator(
            backgroundColor: uiTheme.primaryColor,
          )))
    ],
  );
}
