Called to obtain the widget for each cell of the row.

This method will be called for every row that are visible in datagridâ€™s view port from the collection which is assigned to DataGridSource.rows property.

Return the widgets in the order in which those should be displayed in each column of a row in DataGridRowAdapter.cells.

The number of widgets in the collection must be exactly as many cells as SfDataGrid.columns in the SfDataGrid.

This method will be called whenever you call the notifyListeners method.

@override
DataGridRowAdapter buildRow(DataGridRow row) {
  return DataGridRowAdapter(
      cells: row.getCells().map<Widget>((e) {
    return (e is SmDefaultDataGridCell)
        ? e.enabledMenuOption
            ? buildColumnWithMenuOption(
                onCellClicked: e.onCellTap,
                color: e.dataGridStyle?.bgColor,
                alignment: e.dataGridStyle?.forceAlignment != null &&
                        e.dataGridStyle!.forceAlignment &&
                        e.dataGridStyle!.alignment != null
                    ? e.dataGridStyle!.alignment
                    : _localizationManagerUI.isLTR()
                        ? Alignment.centerLeft
                        : Alignment.centerRight,
                onPress: e.onOptionTap,
                dropDownItems: e.dropDownItems,
                value: e.child != null
                    ? e.child!
                    : _buildDefaultCell(
                        '${e.value}',
                        style: e.style,
                        alignment: e.dataGridStyle?.forceAlignment != null &&
                                e.dataGridStyle!.forceAlignment &&
                                e.dataGridStyle!.alignment != null
                            ? e.dataGridStyle!.alignment
                            : _localizationManagerUI.isLTR()
                                ? Alignment.centerLeft
                                : Alignment.centerRight,
                      ),
              )
            : GestureDetector(
                onTap: e.onCellTap,
                child: Container(
                  color: e.dataGridStyle?.bgColor,
                  alignment: e.dataGridStyle?.forceAlignment != null &&
                          e.dataGridStyle!.forceAlignment &&
                          e.dataGridStyle!.alignment != null
                      ? e.dataGridStyle!.alignment
                      : _localizationManagerUI.isLTR()
                          ? Alignment.centerLeft
                          : Alignment.centerRight,
                  padding: EdgeInsets.all(2.0),
                  child: e.child != null
                      ? e.child
                      : _buildDefaultCell(
                          '${e.value}',
                          style: e.style,
                          alignment:
                              e.dataGridStyle?.forceAlignment != null &&
                                      e.dataGridStyle!.forceAlignment &&
                                      e.dataGridStyle?.alignment != null
                                  ? e.dataGridStyle?.alignment
                                  : _localizationManagerUI.isLTR()
                                      ? Alignment.centerLeft
                                      : Alignment.centerRight,
                        ),
                ),
              )
        : Container(
            color: Colors.white,
            alignment: _localizationManagerUI.isLTR()
                ? Alignment.centerLeft
                : Alignment.centerRight,
            padding: EdgeInsets.all(8.0),
            child: _buildDefaultCell(
              e.value,
              alignment: _localizationManagerUI.isLTR()
                  ? Alignment.centerLeft
                  : Alignment.centerRight,
            ),
          );
  }).toList());
}
