filter(String? columnName, String value, FilterOperator? operator) {
  if (operator == FilterOperator.none ||
      value == null ||
      (value is String && value.isEmpty)) {
    isFiltered = false;
    notifyListeners();
    setSate!(() {});
    return;
  }

  fun(FilterOperator? o, String cellValue) {
    var compare1, compare2;
    if (int.tryParse(cellValue.toString()) == null &&
        DateTime.tryParse(cellValue) != null) {
      compare1 = DateTime.parse(cellValue).millisecondsSinceEpoch;
      compare2 = DateTime.parse(value).millisecondsSinceEpoch;
    } else if (int.tryParse(cellValue) != null) {
      compare1 = int.tryParse(cellValue.toString());
      compare2 = int.tryParse(value.toString());
    } else {
      compare1 = cellValue;
      compare2 = value;
    }
    switch (o) {
      case FilterOperator.equal:
        return compare1 == compare2;
      case FilterOperator.lessThan:
        return compare1 < compare2;
      case FilterOperator.moreThan:
        return compare1 > compare2;
      case FilterOperator.contains:
        return compare1.toString().contains(compare2.toString());
    }
  }

  isFiltered = true;
  _dataGridRows = dataGridRows.where((element) {
    try {
      var res = element.getCells().firstWhere(
            (element) =>
                element.columnName == columnName &&
                fun(operator, element.value.toString()),
          );
      return res != null;
    } catch (e) {
      log("test here ");
    }
    return false;
  }).toList();
  notifyListeners();
  if (setSate != null) setSate!(() {});
}
