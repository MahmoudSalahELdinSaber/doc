Implementation

showSMModalBottomSheet(
  BuildContext context, {
  String? title,
  String? titleWithIcon,
  IconData? close,
  Function? onClose,
  Color? closeIconColor,
  Color? titleColor,
  required Widget content,
  Widget? buttonSection,
  bool hasScrolling = false,
  bool isScrollControlled = false,
  bool isDismissible = true,
  bool enableDrag = true,
  Color? backgroundColor,
  double? height,
  double? horizontalPadding,
}) {
  var sizeConfig = SizeConfig(context);
  return showModalBottomSheet(
      context: context,
      isDismissible: isDismissible,
      enableDrag: enableDrag,
      isScrollControlled: isScrollControlled,
      backgroundColor: backgroundColor ?? uiTheme.surfaceColor,
      shape: RoundedRectangleBorder(
        borderRadius: BorderRadius.only(
            topLeft: Radius.circular(25), topRight: Radius.circular(25)),
      ),
      builder: (context) {
        return Padding(
          padding: EdgeInsets.only(
              left: horizontalPadding ?? horizontalSpace,
              right: horizontalPadding ?? horizontalSpace,
              bottom: MediaQuery.of(context).viewInsets.bottom),
          child: Wrap(
            children: [
              Container(
                alignment: Alignment.center,
                margin: EdgeInsets.only(
                    left: sizeConfig.blockSizeHorizontal * 25,
                    right: sizeConfig.blockSizeHorizontal * 25,
                    top: sizeConfig.blockSizeHorizontal * 2,
                    bottom: sizeConfig.blockSizeHorizontal * 2),
                decoration: BoxDecoration(
                  borderRadius: BorderRadius.circular(12),
                  border: Border.all(color: Color(0xffe1e1e1), width: 2),
                ),
              ),
              Container(
                margin: EdgeInsets.symmetric(vertical: verticalSpace),
                child: title != null
                    ? Text(
                        title,
                        style: TextStyle(
                            fontSize: 20,
                            fontWeight: FontWeight.bold,
                            color: titleColor ?? uiTheme.primaryColor,
                            letterSpacing: 2),
                      )
                    : Container(
                        child: Row(
                          mainAxisAlignment: MainAxisAlignment.spaceBetween,
                          children: [
                            InkWell(
                              onTap: () {
                                if (onClose != null) onClose();
                                Navigator.pop(context);
                              },
                              child: Icon(
                                close ?? Icons.close,
                                color: closeIconColor ?? uiTheme.onsurfaceColor,
                              ),
                            ),
                            Text(
                              titleWithIcon ?? "",
                              style: TextStyle(
                                  color: uiTheme.onsurfaceColor,
                                  fontSize: 18,
                                  fontWeight: FontWeight.bold,
                                  fontFamily: "Cairo"),
                            ),
                          ],
                        ),
                      ),
              ),
              hasScrolling
                  ? Column(
                      mainAxisSize: MainAxisSize.min,
                      children: [
                        Container(
                            height: height ?? sizeConfig.screenHeight! * 0.60,
                            child: content),
                        if (buttonSection != null) buttonSection,
                      ],
                    )
                  : content,
              hasScrolling ? Container() : buttonSection ?? SizedBox.shrink(),
            ],
          ),
        );
      });
}
