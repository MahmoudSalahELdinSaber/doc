TechnicalIndicators getIndicator(SmIndicatorType indicatorType,
    List<SmChartDataModel> items, Color color) {
  switch (indicatorType) {
    case SmIndicatorType.MomentumIndicator:
      return createMomentumIndicator(indicatorColor: color, data: items)!;

    case SmIndicatorType.AccumulationDistributionIndicator:
      return createAccumulationDistributionIndicator(
          indicatorColor: color, data: items)!;
    case SmIndicatorType.AtrIndicator:
      return createAtrIndicator(indicatorColor: color, data: items)!;
    case SmIndicatorType.BollingerBandIndicator:
      return createBollingerBandIndicator(
          indicatorColor: color, data: items)!;
    case SmIndicatorType.EmaIndicator:
      return createEmaIndicator(indicatorColor: color, data: items)!;
    case SmIndicatorType.MacdIndicator:
      return createMacdIndicator(indicatorColor: color, data: items)!;
    case SmIndicatorType.RsiIndicator:
      return createRsiIndicator(indicatorColor: color, data: items)!;
    case SmIndicatorType.SmaIndicator:
      return createSmaIndicator(indicatorColor: color, data: items)!;
    case SmIndicatorType.StochasticIndicator:
      return createSmaIndicator(indicatorColor: color, data: items)!;
    case SmIndicatorType.TmaIndicator:
      return createTmaIndicator(indicatorColor: color, data: items)!;
  }
}
