void setThemeData(
    {PopupMenuThemeData? popupMenuTheme,
    InputDecorationTheme? inputDecorationTheme,
    BottomNavigationBarThemeData? bottomNavigationBarTheme,
    Color? canvasColor,
    Color? scaffoldBackgroundColor,
    Color? unselectedWidgetColor,
    Color? toggleableActiveColor,
    DialogTheme? dialogTheme,
    Color? cardColor,
    TextTheme? textTheme,
    CardTheme? cardTheme,
    ButtonThemeData? buttonTheme,
    IconThemeData? iconTheme,
    AppBarTheme? appBarTheme,
    ColorScheme? colorScheme}) {
  theme = ThemeData(
      popupMenuTheme: popupMenuTheme ??
          PopupMenuThemeData(textStyle: TextStyle(color: onBackgroundColor)),
      inputDecorationTheme: inputDecorationTheme ??
          InputDecorationTheme(
              hintStyle: TextStyle(color: onBackgroundColor),
              filled: true,
              fillColor: Color.alphaBlend(
                  Colors.grey.withOpacity(0.1), backgroundColor),
              iconColor: onBackgroundColor,
              suffixIconColor: onBackgroundColor,
              labelStyle: TextStyle(color: onBackgroundColor),
              helperStyle: TextStyle(color: onBackgroundColor)),
      bottomNavigationBarTheme: bottomNavigationBarTheme ??
          BottomNavigationBarThemeData(
            backgroundColor: backgroundColor,
          ),
      canvasColor: canvasColor ?? backgroundColor,
      scaffoldBackgroundColor: scaffoldBackgroundColor ?? backgroundColor,
      unselectedWidgetColor: unselectedWidgetColor ?? disableColor,
      toggleableActiveColor: toggleableActiveColor ?? primaryColor,
      dialogTheme: dialogTheme ??
          DialogTheme(
              backgroundColor: surfaceColor,
              titleTextStyle: TextStyle(color: onsurfaceColor),
              contentTextStyle: TextStyle(color: onsurfaceColor)),
      cardColor: cardColor ?? surfaceColor,
      textTheme: textTheme ??
          TextTheme(
              bodyText2: TextStyle(color: onBackgroundColor),
              subtitle1: TextStyle(color: onBackgroundColor, fontSize: 16),
              headline2: TextStyle(color: onBackgroundColor, fontSize: 60),
              headline5: TextStyle(color: onBackgroundColor, fontSize: 24),
              headline6: TextStyle(color: onBackgroundColor, fontSize: 20),
              button: TextStyle(color: onBackgroundColor, fontSize: 14)),
      cardTheme: cardTheme ??
          CardTheme(
            color: surfaceColor,
          ),
      buttonTheme: buttonTheme ??
          ButtonThemeData(
              buttonColor: primaryColor, textTheme: ButtonTextTheme.primary),
      iconTheme: iconTheme ?? IconThemeData(color: primaryColor),
      appBarTheme: appBarTheme ??
          AppBarTheme(
              backgroundColor: primaryColor,
              actionsIconTheme: IconThemeData(color: onPrimaryColor),
              titleTextStyle: TextStyle(color: onPrimaryColor, fontSize: 24)),
      colorScheme: colorScheme ??
          ColorScheme(
            background: backgroundColor,
            brightness: Brightness.light,
            surface: onsurfaceColor,
            error: errorColor,
            onBackground: onBackgroundColor,
            onError: errorColor,
            onPrimary: onPrimaryColor,
            onSecondary: onSecondaryColor,
            onSurface: onsurfaceColor,
            primary: primaryColor,
            secondary: secondaryColor,
          ));
}
