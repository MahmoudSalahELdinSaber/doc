static String getFarsiNumber(String s) {
  var sb = new StringBuffer();
  for (int i = 0; i < s.length; i++) {
    switch (s[i]) {
    //Arabic digits
      case '0':
        sb.write('\u0660');
        break;
      case '1':
        sb.write('\u0661');
        break;
      case '2':
        sb.write('\u0662');
        break;
      case '3':
        sb.write('\u0663');
        break;
      case '4':
        sb.write('\u0664');
        break;
      case '5':
        sb.write('\u0665');
        break;
      case '6':
        sb.write('\u0666');
        break;
      case '7':
        sb.write('\u0667');
        break;
      case '8':
        sb.write('\u0668');
        break;
      case '9':
        sb.write('\u0669');
        break;
      default:
        sb.write(s[i]);
        break;
    }
  }
  return sb.toString();
}
